/**
 * Emit the <script> tag for a client-side TS module.
 *
 * Dev mode (DESKWORK_DEV=1): emit /src/<name>.ts so Vite's middleware serves
 * the source with HMR.
 *
 * Prod mode (default): emit /static/dist/<name>.js (the in-process esbuild
 * output served from .runtime-cache/dist/).
 */
export function clientScriptTag(name: string): string {
  if (process.env.DESKWORK_DEV === '1') {
    return `<script type="module" src="/src/${name}.ts"></script>`;
  }
  return `<script type="module" src="/static/dist/${name}.js"></script>`;
}
```

const devMode = process.env.DESKWORK_DEV === '1';

if (devMode) {
  console.log('deskwork-studio: dev mode (DESKWORK_DEV=1) — Vite middleware enabled, esbuild step skipped');
  const { createServer: createViteServer } = await import('vite');
  const vite = await createViteServer({
    server: { middlewareMode: true },
    appType: 'custom',
    root: join(pluginRoot(), 'public', 'src'),
  });
  // Mount Vite's connect middleware on the Hono app.
  // Hono accepts a connect-style middleware via app.use with c.env.incoming/outgoing.
  app.use(async (c, next) => {
    const req = c.env.incoming;
    const res = c.env.outgoing;
    if (!req || !res) {
      await next();
      return;
    }
    return new Promise<void>((resolve) => {
      vite.middlewares(req, res, () => {
        next().then(() => resolve());
      });
    });
  });
} else {
  // Prod: existing in-process esbuild step.
  const summary = await buildClientAssets({ pluginRoot: pluginRoot() });
  console.log(`deskwork-studio: built ${summary.built} client assets (${summary.cached} cached) -> ${summary.outDir}`);
}
```

Notes for the implementer:
- Hono's middleware integration with connect-style middleware varies by adapter. The `@hono/node-server` adapter exposes `c.env.incoming` and `c.env.outgoing` (the raw `IncomingMessage` and `ServerResponse`). If those don't exist on the current adapter, fall back to: skip Vite's `app.use` and instead register a route `app.all('/src/*', ...)` and `app.all('/@vite/*', ...)` that pipes to `vite.middlewares` directly.
- The `vite` import is dynamic (`await import('vite')`) so vite is not loaded in prod (where it's not in the npm install set).

import { readdir } from 'node:fs/promises';

async function findIngestSourceFile(
  projectRoot: string,
  entryId: string,
): Promise<string | undefined> {
  const ingestDir = join(projectRoot, '.deskwork', 'review-journal', 'ingest');
  let files: string[];
  try {
    files = await readdir(ingestDir);
  } catch {
    return undefined;
  }
  for (const f of files) {
    if (!f.endsWith('.json')) continue;
    try {
      const body = await readFile(join(ingestDir, f), 'utf8');
      const rec: { entryId?: string; sourceFile?: string } = JSON.parse(body);
      if (rec.entryId === entryId && typeof rec.sourceFile === 'string') {
        return rec.sourceFile;
      }
    } catch {
      // skip malformed
    }
  }
  return undefined;
}

async function readLegacyPipelineRecords(
  projectRoot: string,
): Promise<Map<string, { currentVersion: number; state: string; stage: Stage }>> {
  const dir = join(projectRoot, '.deskwork', 'review-journal', 'pipeline');
  const out = new Map<string, { currentVersion: number; state: string; stage: Stage }>();
  let files: string[];
  try {
    files = await readdir(dir);
  } catch {
    return out;
  }
  for (const f of files) {
    if (!f.endsWith('.json')) continue;
    try {
      const body = await readFile(join(dir, f), 'utf8');
      const rec: {
        entryId?: string;
        state?: string;
        currentVersion?: number;
      } = JSON.parse(body);
      if (typeof rec.entryId !== 'string') continue;
      if (typeof rec.currentVersion !== 'number') continue;
      if (typeof rec.state !== 'string') continue;
      const existing = out.get(rec.entryId);
      // Keep highest currentVersion if multiple records exist per entry.
      if (!existing || rec.currentVersion > existing.currentVersion) {
        // Stage is read from sidecar later; placeholder here.
        out.set(rec.entryId, { currentVersion: rec.currentVersion, state: rec.state, stage: 'Drafting' });
      }
    } catch {
      // skip malformed
    }
  }
  return out;
}

  toStage: Stage;

    entryId: sidecar.uuid,
    from,
    to,
  });
  return { entryId: sidecar.uuid, fromStage: from, toStage: to };
